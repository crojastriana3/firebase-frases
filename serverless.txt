-Reglas de seguridad: Estas reglas nos permiten indicar quien escribe y quien lee nustra base de datos.
-rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId}{
    allow create: if request.auth.uid != null; //si no esta crea cuenta
    allow read: if request.auth.uid == userId; //si esta lee tus datos
    }
    match /frases/{frasesId}{
    allow read, write: if request.auth.uid != null; //todo user loguado puede leer y escribir
    }
  }
}
